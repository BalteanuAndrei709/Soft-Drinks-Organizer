CREATE TABLE newsletter (
    	id int NOT NULL PRIMARY KEY AUTO_INCREMENT,
    	email varchar(100) NOT NULL UNIQUE
);

CREATE TABLE users (
	id int(100) PRIMARY KEY NOT NULL AUTO_INCREMENT,
	TYPE varchar(6) NOT NULL DEFAULT 'user,
	username TINYTEXT NOT NULL UNIQUE,
	email TYNYTEXT NOT NULL UNIQUE,
	password LONGTEXT NOT NULL,
	fullname varchar(100) NOT NULL,
	description TEXT,
	image varchar(300) DEFAULT 'basic-profile-picture.jpg'
);
		
CREATE TABLE products (
	id int NOT NULL PRIMARY KEY AUTO_INCREMENT,
	name varchar(100) NOT NULL,
	price float,
	size varchar(50),
	calories float,
	fat float,
	carbs float,
	fiber float,
	protein float,
	salt float,
	ingredients varchar(500),
	category_id int
);

CREATE TABLE categories (
  	id int NOT NULL PRIMARY KEY AUTO_INCREMENT,
  	name varchar(100)
);

ALTER TABLE products ADD FOREIGN KEY (category_id) REFERENCES categorie (id);
